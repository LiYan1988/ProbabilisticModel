1function [ DemandStruct ] = modifyDemandStruct(DemandStruct)
 2% Create DemandStruct is too long for large networks. So we create it once
 3% and modify it when it is needed. The node pairs will not be changed, only
 4% the bandwidth requirement will be changed.
 5% 
 6
 7% Generate new traffic demands
 8Ndemands = size(DemandStruct.demandsMatrix, 1);
 9
10demandDataRate = zeros(Ndemands, 1);
11if strcmp(DemandStruct.distribution, 'uniform')
12    demandDataRate = randi([DemandStruct.distributionParameter1, ...
13        DemandStruct.distributionParameter2], ...
14        [Ndemands, 1]);
15elseif strcmp(DemandStruct.distribution, 'normal')
16    % DataRateLowerBound is mean, DataRateUpperBound is std
17    demandDataRate = round(normrnd(DemandStruct.distributionParameter1, ...
18        DemandStruct.distributionParameter2, ...
19        [Ndemands, 1]));
20end
21DemandStruct.demandsMatrix(:, 3) = demandDataRate;
22% DemandStruct.demandsTable.DataRate = demandDataRate;
