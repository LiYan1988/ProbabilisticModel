1function [ noise_all, noise_sci, noise_xci, noise_ase, noise_xci_ub ] = ...
 2    calculateNoise(demandsBandwidth, demandsCenterFrequency, psd, ...
 3    Nspan, alpha, beta, gamma, Nase, gb)
 4% Calculate noise for all the demands on a single link, according to
 5% the PTL paper.
 6%
 7% Units of input parameters:
 8% psd: muW/GHz ~ 10e-15 W/Hz, typical value: 15 muW/GHz
 9% bandwidth and center frequency: GHz, typical value: 100 GHz
10%
11% Corresponding constants:
12% mu: 7.58e-7 (muW/GHz)^(-2)
13% rho: 2.11e-3 (GHz)^(-2)
14% Nase: 3.58e-2 (muW/GHz)
15
16Nuser = length(demandsBandwidth);
17x_b = demandsBandwidth;
18x_f = demandsCenterFrequency;
19
20mu = 3*gamma^2/(2*pi*alpha*abs(beta));
21rho = pi^2*abs(beta)/(2*alpha);
22mu = mu*1e-30;
23rho = rho*1e18;
24Nase = Nase*1e15;
25
26% ASE noise
27noise_ase = Nspan*Nase;
28
29% SCI component: noise_sci(i) = psd_i^2*asinh(rho*Df_i^2)
30noise_sci = Nspan*mu*psd.^3.*asinh(rho*x_b.^2);
31
32% XCI component
33f_mat = repmat(x_f.', Nuser, 1); % f_mat_{i, j} = f_j
34f_mat = abs(f_mat.'-f_mat); % f_mat_{i, j} = |f_i-f_j|
35xci_numerator = f_mat+0.5*repmat(x_b.', Nuser, 1);
36% |f_j-f_i|+0.5*Df_j
37xci_denominator = f_mat-0.5*repmat(x_b.', Nuser, 1);
38% |f_j-f_i|-0.5*Df_j
39p_mat = repmat((psd.').^2, Nuser, 1).*...
40    log(abs(xci_numerator./xci_denominator));
41% p_mat(i, j) = psd_j^2*log((|f_j-f_i|+0.5*Df_j)/(|f_j-f_i|-0.5*Df_j))
42noise_xci = Nspan*mu*psd.*sum(p_mat, 2);
43
44noise_all = noise_ase+noise_sci+noise_xci;
45
46% estimate xci
47psd_max = max(psd);
48bandwidth_total = sum(x_b);
49noise_xci_ub = 2*mu*Nspan*psd_max^3*log((bandwidth_total+2*gb)./(x_b+2*gb));
