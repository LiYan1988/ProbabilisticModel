1function [blockFlag, demandsNoisePerLinkAll, demandsFrequency] = ...
  2    checkBlock(idx, slotStart, demandsFrequency, ...
  3    demandsNoisePerLinkAll, systemParameters, TopologyStruct, ...
  4    DemandStruct, SetOfDemandsOnLink, SetOfDemandsOnNode, ...
  5    demandPaths, demandPathLinks)
  6% Check if the new demand causes higher than threshold noie to existing
  7% demands or itself.
  8
  9alpha = systemParameters.alpha;
 10beta = systemParameters.beta;
 11gamma = systemParameters.gamma;
 12Nase = systemParameters.Nase;
 13gb = systemParameters.gb;
 14psd = systemParameters.psd;
 15modulationFormat = systemParameters.modulationFormat;
 16if strcmp(modulationFormat, 'PM_QPSK')
 17    NoiseMax = systemParameters.psd/systemParameters.snrThresholds.('PM_QPSK');
 18elseif strcmp(modulationFormat, 'PM_8QAM')
 19    NoiseMax = systemParameters.psd/systemParameters.snrThresholds.('PM_8QAM');
 20elseif strcmp(modulationFormat, 'PM_16QAM')
 21    NoiseMax = systemParameters.psd/systemParameters.snrThresholds.('PM_16QAM');
 22elseif strcmp(modulationFormat, 'PM_32QAM')
 23    NoiseMax = systemParameters.psd/systemParameters.snrThresholds.('PM_32QAM');
 24else
 25    NoiseMax = systemParameters.psd/systemParameters.snrThresholds.('PM_64QAM');
 26end
 27
 28LinkLengths = TopologyStruct.LinkLengths;
 29RS = TopologyStruct.RegenSites;
 30
 31demandsMatrix = DemandStruct.demandsMatrix;
 32demandPathNodes = demandPaths;
 33
 34% pretend the new demand is allocated
 35demandsFrequencyTmp = demandsFrequency;
 36demandsFrequencyTmp(idx, 1) = slotStart(1)-1+demandsMatrix(idx, 3)/2;
 37demandsFrequencyTmp(idx, 2) = slotStart(1)-1;
 38demandsFrequencyTmp(idx, 3) = slotStart(1)-1+demandsMatrix(idx, 3);
 39
 40demandsNoisePerLinkAllTmp = demandsNoisePerLinkAll;
 41
 42% calculate the noise on each link used by the new demand
 43demandsInvoled = zeros(1, size(demandsMatrix, 1)); % the demands need to check
 44nz = 0;
 45linksUsed = demandPathLinks(idx, 1:demandPathLinks(idx, end));
 46for i=1:length(linksUsed)
 47    demandsOnLink = SetOfDemandsOnLink(linksUsed(i), 1:SetOfDemandsOnLink(linksUsed(i), end));
 48    if isempty(demandsOnLink)
 49        continue;
 50    end
 51    % remove demands that do not use this link
 52    for j=1:length(demandsOnLink)
 53        if demandsFrequencyTmp(demandsOnLink(j), 3)==0
 54            demandsOnLink(j) = -1;
 55        end
 56    end
 57    demandsOnLink(demandsOnLink==-1) = [];
 58    if isempty(demandsOnLink)
 59        continue;
 60    end
 61    % demandsInvoled = union(demandsInvoled, demandsOnLink, 'rows');
 62    tmp = demandsOnLink;
 63    for n = 1:nz
 64        flag = 0;
 65        for m = 1:length(demandsOnLink)
 66            if demandsInvoled(n)==demandsOnLink(m)
 67                flag = 1;
 68                break;
 69            end
 70        end
 71        if flag==0
 72            tmp0 = [tmp, demandsInvoled(n)];
 73            tmp = tmp0;
 74        end
 75    end
 76    demandsInvoled(1:length(tmp)) = tmp;
 77    nz = length(tmp);
 78    
 79    demandsCenterFrequency = demandsFrequencyTmp(demandsOnLink, 1);
 80    demandsBandwidth = demandsMatrix(demandsOnLink, 3);
 81    demandsPSD = psd*ones(length(demandsOnLink), 1);
 82    [noise_all, noise_sci, noise_xci, noise_ase, noise_xci_ub] = ...
 83        calculateNoise(demandsBandwidth, demandsCenterFrequency, ...
 84        demandsPSD, LinkLengths(i), alpha, beta, gamma, Nase, gb);
 85    demandsNoisePerLinkAllTmp(demandsOnLink, linksUsed(i)) = noise_all;
 86end
 87
 88% check if any demand is worse than nosie threshold
 89blockFlag = false;
 90for d=1:nz
 91    linkTmp = demandPathLinks(demandsInvoled(d), 1:demandPathLinks(demandsInvoled(d), end));
 92    nodeTmp = demandPathNodes(demandsInvoled(d), 1:demandPathNodes(demandsInvoled(d), end));
 93    noiseTmp = 0; % accumulated noie
 94    for l=1:length(linkTmp)
 95        noiseTmp = noiseTmp+demandsNoisePerLinkAllTmp(demandsInvoled(d), linkTmp(l));
 96        if noiseTmp > NoiseMax
 97            blockFlag = true;
 98            break
 99        end
100        if ismember(nodeTmp(l+1), RS)
101            noiseTmp = 0;
102        end
103    end
104    if blockFlag
105        break
106    end
107end
108
109if ~blockFlag
110    demandsNoisePerLinkAll = demandsNoisePerLinkAllTmp;
111    demandsFrequency = demandsFrequencyTmp;
112end
