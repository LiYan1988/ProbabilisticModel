1function [blockStatistics, blockHistory] = ...
 2    allocateOneByOneBP(systemParameters, TopologyStruct, ...
 3    DemandStruct, demandOrder, ...
 4    SetOfDemandsOnLink, SetOfDemandsOnNode, demandPaths, demandPathLinks)
 5% allocate demands in network one by one
 6% Output:
 7% demandsFrequency: center/start/end frequency
 8%
 9%
10
11% extract parameters
12gb = systemParameters.gb;
13freqMax = systemParameters.freqMax;
14
15NLinks = TopologyStruct.NLinks;

19
20% allocate demands one by one
21blockHistory = zeros(Ndemands, 1);
22blockStatistics = zeros(Ndemands, 1);
23demandsFrequency = zeros(Ndemands, 3);
24frequencySlotsAvailability = ones(freqMax, NLinks); % 1 is usable
25demandsNoisePerLinkAll = zeros(Ndemands, NLinks);

30    temp1 = ones(freqMax, 1);
31    for j=1:length(linkUsed)
32        temp1 = temp1.*frequencySlotsAvailability(:, linkUsed(j));
33    end
34    temp2 = [0; temp1; 0];
35    temp2 = diff(temp2);
36    run_start = find(temp2==1);
37    run_end = find(temp2==-1);
38    run_interval = run_end-run_start;
39    run_idx = find(run_interval>=demandsMatrix(idx, 3)+gb, 1, 'first');
40    run_idx = run_idx(1);
41    % continue if there is no new demand
42    if isempty(run_idx)
43        % block demand i
44        blockStatistics(idx) = 1;
45%         fprintf('Demand %d is blocked due to wavelength, block probability %.4e, %d allocated.\n', idx, sum(blockStatistics)/Ndemands, i);
46        continue
47    end
48    % check noise for the new and exist demands
49    slotStart = run_start(run_idx);
50    [blockFlag, demandsNoisePerLinkAll, demandsFrequency] = ...
51        checkBlock(idx, slotStart, demandsFrequency, ...
52        demandsNoisePerLinkAll, systemParameters, TopologyStruct, ...
53        DemandStruct, ...
54        SetOfDemandsOnLink, SetOfDemandsOnNode, demandPaths, demandPathLinks);
55    if ~blockFlag
56        % if the demand is not blocked, allocate resource
57        for j=1:length(linkUsed)
58            frequencySlotsAvailability(run_start(run_idx):...
59                run_start(run_idx)+demandsMatrix(idx, 3)-1+gb, ...
60                linkUsed(j)) = 0;
61        end
62%         demandsFrequency(idx, 1) = run_start(run_idx)-1+demandsMatrix(idx, 3)/2;
63%         demandsFrequency(idx, 2) = run_start(run_idx)-1;
64%         demandsFrequency(idx, 3) = run_start(run_idx)-1+demandsMatrix(idx, 3);
65    else
66        % else block it
67        blockStatistics(idx) = 1;
68        blockHistory(i:end) = sum(blockStatistics)/Ndemands;
69%         fprintf('Demand %d is blocked due to SNR, block probability %.4e, %d allocated.\n', idx, sum(blockStatistics)/Ndemands, i);
70    end
71end
