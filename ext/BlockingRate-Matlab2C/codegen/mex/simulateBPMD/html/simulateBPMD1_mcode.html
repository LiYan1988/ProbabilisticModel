1function [blockStatistics, blockHistory] = simulateBPMD(randomSeed, ...
  2    simuIDstr, modulationFormat, alg, nRS, simulationName, NMonteCarlo)
  3%% Template file for the journal paper BP simulation on Rivanna
  4% warning off;
  5% addpath(genpath('/scratch/ly6j/YALMIP'));
  6% addpath(genpath('/share/apps/gurobi/6.5.1/matlab'));
  7
  8% simuID = 1;
  9rng(randomSeed);
 10
 11%% Define fiber parameters
 12alpha = 0.22; % dB/km, attenuation of fiber, NOTE: alpha is positive!
 13alpha = alpha*1e-4*log(10); % 1/m
 14L = 100e3; % m, length of one span
 15h = 6.626e-34; % J*s, Plank's constant
 16niu = 193.548e12; % Hz, frequency of lightwave at 1550 nm
 17nsp = 10^(5.5/10)/2; % spontaneous emission factor
 18Nase = (exp(alpha*L)-1)*h*niu*nsp; % ASE per polarization per span
 19% W/Hz, signal side ASE noise spectral density
 20gamma = 1.32e-3; % 1/(W*m), nonlinear parameter
 21% gamma = 0;
 22beta = -2.1668e-26; % s^2/m, GVD parameter, D = 18 ps/(nm*km),
 23% beta = -D*lambda^2/(2*pi*c)
 24beta = abs(beta); % the absolute value is used in calculation
 25
 26systemParameters = struct();
 27systemParameters.alpha = alpha;
 28systemParameters.beta = beta;
 29systemParameters.gamma = gamma;
 30systemParameters.Nase = Nase;
 31systemParameters.snrThresholds = struct('PM_BSPK', 3.52, ...
 32    'PM_QPSK', 7.03, 'PM_8QAM', 17.59, 'PM_16QAM', 32.60, ...
 33    'PM_32QAM', 64.91, 'PM_64QAM', 127.51);
 34
 35%####################################################################
 36systemParameters.gb = 13; % the guardband
 37systemParameters.freqMax = 320000; % max frequency in GHz
 38systemParameters.psd = 15;
 39systemParameters.modulationFormat = modulationFormat; %'PM_QPSK';
 40systemParameters.Cmax = 100;
 41systemParameters.CircuitWeight = 0.1;
 42systemParameters.RegenWeight = 1;
 43systemParameters.outageProb = 0.0001;
 44%####################################################################
 45
 46%% topology
 47% German network
 48coronet = load('CoronetTopology.mat');
 49NetworkCost = coronet.networkCostMatrix;
 50
 51NNodes = size(NetworkCost, 1);
 52NodeList = 1:NNodes;
 53NetworkConnectivity = 1-isinf(NetworkCost);
 54tmpNetworkCost = NetworkConnectivity.*NetworkCost;
 55tmpNetworkCost(isnan(tmpNetworkCost)) = 0;
 56[i, j, s] = find(tmpNetworkCost);
 57LinkList = [i, j];
 58NLinks = size(LinkList, 1);
 59LinkListIDs = (1:NLinks)';
 60LinkLengths = s;
 61
 62TopologyStruct = struct();
 63TopologyStruct.NodeList = NodeList;
 64TopologyStruct.NNodes = NNodes;
 65TopologyStruct.NetworkCost = NetworkCost;
 66TopologyStruct.NetworkConnectivity = NetworkConnectivity;
 67TopologyStruct.LinkList = LinkList;
 68TopologyStruct.NLinks = NLinks;
 69TopologyStruct.LinkListIDs = LinkListIDs;
 70TopologyStruct.LinkLengths = LinkLengths;
 71
 72%% generate traffic demands
 73%####################################################################
 74Ndemands = 5550; % don't need to change
 75Nsamples = 1; % don't need to change
 76distributionName = 'normal';
 77p1 = 150;
 78p2 = 20;
 79ndprob=0.8;
 80ndmax=3;
 81% NMonteCarlo = 1;
 82Repeat = 1;
 83Nbins = 65;
 84Mbins = 50;
 85Sbins = 15;
 86% RoutingScheme = 'MD';
 87% alg = 'benchmark_Bathula_MD_subopt'; % or 'proposed1', 'proposed2'
 88% nRS = 20;
 89% simulationName = 'template';
 90%####################################################################
 91
 92SimulationParameters = struct();
 93SimulationParameters.p1 = p1;
 94SimulationParameters.p2 = p2;
 95SimulationParameters.ndprob = ndprob;
 96SimulationParameters.ndmax = ndmax;
 97SimulationParameters.distributionName = distributionName;
 98SimulationParameters.NMonteCarlo = NMonteCarlo;
 99SimulationParameters.Repeat = Repeat;
100SimulationParameters.Ndemands = Ndemands;
101SimulationParameters.Nsamples = Nsamples;
102SimulationParameters.Nbins = Nbins;
103SimulationParameters.Mbins = Mbins;
104SimulationParameters.Sbins = Sbins;
105
106%% Monte Carlo
107x = load('templateDemandStructMDForC.mat');
108DemandStructTemplate = x.DemandStructMD;
109% clear x;
110DemandStructTemplate.distributionParameter1 = p1;
111DemandStructTemplate.distributionParameter2 = p2;
112
113% proposed algorithms
114if strcmp(alg, 'proposed_RS_MD') % min-dist, RS ranking
115    x = load('RSMD1.mat');
116    RS = x.Ii_sort_idx;
117    TopologyStruct.RegenSites = RS(1:nRS);
118elseif strcmp(alg, 'proposed_RC_MD') % min-dist, RC ranking
119    x = load('RSMD2.mat');
120    RS = x.Cn_sort_idx;
121    TopologyStruct.RegenSites = RS(1:nRS);
122elseif strcmp(alg, 'benchmark_Bathula_MD') % min-dist, Bathula's MILP
123    x = load('RS_MILP_TR2000.mat');
124    RS = x.RS_min_dist;
125    TopologyStruct.RegenSites = RS;
126elseif strcmp(alg, 'benchmark_Bathula_MD_subopt') % min-dist, Bathula's MILP
127    x = load('RS_MILP_TR2000.mat');
128    RS = x.RS_min_dist_subopt;
129    TopologyStruct.RegenSites = RS;
130elseif strcmp(alg, 'benchmark_Predo_routingOnly') % routing only, Predo's 
131    x = load('PredoRoutingOnly.mat');
132    RS = x.NodeProbROSortedIdx;
133    TopologyStruct.RegenSites = RS(1:nRS);
134% elseif strcmp(alg, 'benchmark_Predo_routingReach') % routing only, Predo's 
135else
136    x = load('PredoRoutingReach.mat');
137    RS = x.NodeProbRRSortedIdx;
138    TopologyStruct.RegenSites = RS(1:nRS);
139end
140
141cellMatrices = load('CellMatricesMD.mat');
142SetOfDemandsOnLink = cellMatrices.SetOfDemandsOnLinkMatrix;
143SetOfDemandsOnNode = cellMatrices.SetOfDemandsOnNodeMatrix;
144demandPathLinks = cellMatrices.demandPathLinksMatrix;
145demandPaths = cellMatrices.demandPathsMatrix;
146
147blockHistory = zeros(Ndemands, NMonteCarlo);
148blockStatistics = zeros(Ndemands, NMonteCarlo);
149[blockStatistics, blockHistory] = ...
150    simulateBlockProb(systemParameters, TopologyStruct, ...
151    SimulationParameters, DemandStructTemplate, ...
152    SetOfDemandsOnLink, SetOfDemandsOnNode, demandPaths, demandPathLinks);
153
154fileName = [simulationName, '_', alg, '_MD_blockHistory_', simuIDstr, '.csv'];
155% fileName = sprintfc('%s_%s_MD_blockHistory_%d.csv', simulationName, alg, int32(simuID));
156fid = fopen(fileName, 'w');
157for d = 1:Ndemands
158    for n = 1:NMonteCarlo
159        fprintf(fid, '%.8e,', blockHistory(d, n));
160    end
161    fprintf(fid, '\n');
162end
163fclose(fid);
