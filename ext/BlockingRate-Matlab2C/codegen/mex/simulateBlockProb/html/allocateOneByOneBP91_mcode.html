1function [blockStatistics, blockHistory] = ...
 2    allocateOneByOneBP(systemParameters, TopologyStruct, ...
 3    DemandStruct, demandOrder, ...
 4    SetOfDemandsOnLink, SetOfDemandsOnNode, demandPaths, demandPathLinks)
 5% allocate demands in network one by one
 6% Output:
 7% demandsFrequency: center/start/end frequency
 8%
 9%
10
11% extract parameters
12gb = systemParameters.gb;

19

23demandsFrequency = zeros(Ndemands, 3);
24frequencySlotsAvailability = ones(freqMax, NLinks); % 1 is usable
25demandsNoisePerLinkAll = zeros(Ndemands, NLinks);

30    temp1 = ones(freqMax, 1);
31    for j=1:length(linkUsed)
32        temp1 = temp1.*frequencySlotsAvailability(:, linkUsed(j));
33    end
34    temp2 = [0; temp1; 0];
35    temp2 = diff(temp2);
36    run_start = find(temp2==1);
37    run_end = find(temp2==-1);
38    run_interval = run_end-run_start;
39    run_idx = find(run_interval>=demandsMatrix(idx, 3)+gb, 1, 'first');
40    % continue if there is no new demand
41    if isempty(run_idx)
42        % block demand i
43        blockStatistics(idx) = 1;
44%         fprintf('Demand %d is blocked due to wavelength, block probability %.4e, %d allocated.\n', idx, sum(blockStatistics)/Ndemands, i);
45        continue
46    end
47    % check noise for the new and exist demands
48    slotStart = run_start(run_idx);
49    [blockFlag, demandsNoisePerLinkAll, demandsFrequency] = ...
50        checkBlock(idx, slotStart, demandsFrequency, ...
51        demandsNoisePerLinkAll, systemParameters, TopologyStruct, ...
52        DemandStruct, ...
53        SetOfDemandsOnLink, SetOfDemandsOnNode, demandPaths, demandPathLinks);
54    if ~blockFlag
55        % if the demand is not blocked, allocate resource
56        for j=1:length(linkUsed)
57            frequencySlotsAvailability(run_start(run_idx):...
58                run_start(run_idx)+demandsMatrix(idx, 3)-1+gb, ...
59                linkUsed(j)) = 0;
60        end
61%         demandsFrequency(idx, 1) = run_start(run_idx)-1+demandsMatrix(idx, 3)/2;
62%         demandsFrequency(idx, 2) = run_start(run_idx)-1;
63%         demandsFrequency(idx, 3) = run_start(run_idx)-1+demandsMatrix(idx, 3);
64    else
65        % else block it
66        blockStatistics(idx) = 1;
67        blockHistory(i:end) = sum(blockStatistics)/Ndemands;
68%         fprintf('Demand %d is blocked due to SNR, block probability %.4e, %d allocated.\n', idx, sum(blockStatistics)/Ndemands, i);
69    end
70end
