1function [blockFlag, demandsNoisePerLinkAll, demandsFrequency] = ...
 2    checkBlock(idx, slotStart, demandsFrequency, ...
 3    demandsNoisePerLinkAll, systemParameters, TopologyStruct, ...
 4    DemandStruct, SetOfDemandsOnLink, SetOfDemandsOnNode, ...
 5    demandPaths, demandPathLinks)
 6% Check if the new demand causes higher than threshold noie to existing
 7% demands or itself.
 8
 9alpha = systemParameters.alpha;
10beta = systemParameters.beta;
11gamma = systemParameters.gamma;
12Nase = systemParameters.Nase;
13gb = systemParameters.gb;
14psd = systemParameters.psd;
15NoiseMax = systemParameters.psd/systemParameters.snrThresholds.(systemParameters.modulationFormat);
16
17LinkLengths = TopologyStruct.LinkLengths;
18RS = TopologyStruct.RegenSites;
19
20demandsMatrix = DemandStruct.demandsMatrix;
21demandPathNodes = demandPaths;
22
23% pretend the new demand is allocated
24demandsFrequencyTmp = demandsFrequency;
25demandsFrequencyTmp(idx, 1) = slotStart-1+demandsMatrix(idx, 3)/2;
26demandsFrequencyTmp(idx, 2) = slotStart-1;
27demandsFrequencyTmp(idx, 3) = slotStart-1+demandsMatrix(idx, 3);
28
29demandsNoisePerLinkAllTmp = demandsNoisePerLinkAll;
30
31% calculate the noise on each link used by the new demand
32demandsInvoled = []; % the demands need to check
33linksUsed = demandPathLinks{idx};
34for i=1:length(linksUsed)
35    demandsOnLink = SetOfDemandsOnLink{linksUsed(i)};
36    if isempty(demandsOnLink)
37        continue;
38    end
39    % remove demands that do not use this link
40    for j=1:length(demandsOnLink)
41        if demandsFrequencyTmp(demandsOnLink(j), 3)==0
42            demandsOnLink(j) = -1;
43        end
44    end
45    demandsOnLink(demandsOnLink==-1) = [];
46    if isempty(demandsOnLink)
47        continue;
48    end
49    demandsInvoled = union(demandsInvoled, demandsOnLink);
50    
51    demandsCenterFrequency = demandsFrequencyTmp(demandsOnLink, 1);
52    demandsBandwidth = demandsMatrix(demandsOnLink, 3);
53    demandsPSD = psd*ones(length(demandsOnLink), 1);
54    [noise_all, noise_sci, noise_xci, noise_ase, noise_xci_ub] = ...
55        calculateNoise(demandsBandwidth, demandsCenterFrequency, ...
56        demandsPSD, LinkLengths(i), alpha, beta, gamma, Nase, gb);
57    demandsNoisePerLinkAllTmp(demandsOnLink, linksUsed(i)) = noise_all;
58end
59
60% check if any demand is worse than nosie threshold
61blockFlag = false;
62for d=1:length(demandsInvoled)
63    linkTmp = demandPathLinks{demandsInvoled(d)};
64    nodeTmp = demandPathNodes{demandsInvoled(d)};
65    noiseTmp = 0; % accumulated noie
66    for l=1:length(linkTmp)
67        noiseTmp = noiseTmp+demandsNoisePerLinkAllTmp(demandsInvoled(d), linkTmp(l));
68        if noiseTmp > NoiseMax
69            blockFlag = true;
70            break
71        end
72        if ismember(nodeTmp(l+1), RS)
73            noiseTmp = 0;
74        end
75    end
76    if blockFlag
77        break
78    end
79end
80
81if ~blockFlag
82    demandsNoisePerLinkAll = demandsNoisePerLinkAllTmp;
83    demandsFrequency = demandsFrequencyTmp;
84end
